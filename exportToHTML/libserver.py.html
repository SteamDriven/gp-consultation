28           if self.server_socket:
29               self.server_socket.close()

35       @staticmethod
36       def handle_password(password): # This method will hash a given password and return the hash
37           client_password = password
38           hash_alg = hashlib.new("SHA256")
39   
40           hash_alg.update(client_password.encode())
41           client_hash = hash_alg.hexdigest()
42   
43           print(client_hash)
44           return client_hash

47           self.message["COMMAND"] = " "  # Reset the message dict to prepare for future messages
48           if data:
49               self.query = ""

65                   self.query_data.extend(

97       @staticmethod
98       def generate_code(size): # Used to generate an alphanumeric code at any provided length
99           length = size
100          chars = (string.ascii_uppercase + string.digits)
101          code = []
102  
103          for c in range(length):
104              code.append(random.choice(chars))
105  
106          code = ''.join(code)
107          return code

116      def handle_messages_from_client(self, client): # This will deal with all client requests and return information back
117          self.message = {'COMMAND': "", 'CLIENT': "", 'DATA': " "} # Default template used to send information back to the client
118  
119          users = {-1: None, 1: 'PATIENT', 2: 'CLINICIAN'}
120  
121          while True:
122              data = client.recv(2048)  # Receive information from the client within a given number of bits
123              if not data:
124                  continue
125              else:
126                  message = json.loads(data.decode()) # Decode the encoded data sent from the client
127  
128                  if message['COMMAND'] == 'REGISTER':
129                      print(">: Client requested to be registered to database.")
130  
131                      if self.register_user(message["DATA"], message['CLIENT']):
132                          self.message["COMMAND"] = 'COMPLETED'
133                          self.client_connected.send(json.dumps(self.message).encode())
134  
135                  if message['COMMAND'] == 'REFERRAL':
136                      print(">: Client requested a referral code from server.")
137                      code = self.generate_code(6)
138  
139                      self.message["COMMAND"] = 'REFERRAL'
140                      self.message["CLIENT"] = 'CLINICIAN'
141                      self.message["DATA"] = code
142  
143                      self.client_connected.send(json.dumps(self.message).encode())
144  
145                  if message['COMMAND'] == 'VALIDATE REGISTER':
146                      found_results = self.database.check_records(message)
147                      print(found_results)
148  
149                      if found_results:
150                          self.message['COMMAND'] = 'FAILED'
151                          print(self.message)
152                          self.client_connected.send(json.dumps(self.message).encode())
153  
154                      elif not found_results:
155                          self.message['COMMAND'] = 'PASSED'
156                          print(self.message)
157                          self.client_connected.send(json.dumps(self.message).encode())
158  
159                  if message['COMMAND'] == 'LOGIN':
160                      print(">: Server received request to validate login credentials.")
161                      accept_login = self.compare_login(message)
162                      print(users[accept_login[0]])
163  
164                      if users[accept_login[0]] is not None:
165                          print(f">: User is a {users[accept_login[0]]}")
166  
167                          self.message['COMMAND'] = 'ACCEPT'
168                          self.message['CLIENT'] = users[accept_login[0]]
169                          self.message['DATA'] = accept_login[1]
170                          print(self.message)
171  
172                          self.client_connected.send(json.dumps(self.message).encode())
173  
174                      else:
175                          self.message['COMMAND'] = 'DENY'
176                          self.message['CLIENT'] = users[accept_login[0]]
177                          self.message['DATA'] = None
178                          print(self.message)
179                          self.client_connected.send(json.dumps(self.message).encode())
180  
181      def start(self): # Creates the connection to the socket 
182          try:
183              self.server_socket.bind((self.host, self.port))
184              self.server_socket.listen()
185              print(f">: Server is successfully setup and listening from port: {self.port}")
186  
187              while True: # Thread continously listens for incoming requests from the client but is stopped by the accept blocker.
188                  self.client_connected, self.client_address = self.server_socket.accept()
189                  print(f">: Accepted connection by Address: {self.client_address}")
190                  threading.Thread(target=self.handle_messages_from_client, args=(self.client_connected,)).start()
191  
192          except socket.error as err:
193              print(err)
194              print(f">: [SERVER]: Failed to connect to socket on HOST: {self.host}")
